Set-Content backend\Dockerfile @"
# Build stage
FROM golang:1.22-alpine AS build
WORKDIR /src
COPY go.mod ./
RUN go mod download
COPY . .
RUN CGO_ENABLED=0 GOOS=linux GOARCH=amd64 go build -o /out/api ./cmd/api

# Run stage
FROM alpine:3.20
RUN adduser -D -H -s /sbin/nologin appuser
WORKDIR /app
COPY --from=build /out/api /app/api
USER appuser
EXPOSE 8080
ENV PORT=8080
CMD ["/app/api"]
"@
